16  package org.devacfr.maven.skins.reflow.model;
17  
18  import javax.annotation.Nonnull;
19  
20  import org.devacfr.maven.skins.reflow.ISkinConfig;
21  
22  /**
23   * @author devacfr
24   * @since 2.0
25   */
26  public class TocTopBar extends Toc<TocTopBar> {
27  
28      /** no flatten attribute (default) - flatten if single top heading */
29      private boolean flatten = false;
30  
31      /** -1 or no numberItems attribute (default) - unlimited */
32      private int numberItems = -1;
33  
34      /**
35       * Default constructor.
36       *
37       * @param config
38       *            a config (can <b>not</b> be {@code null}).
39       */
40      public TocTopBar(final @Nonnull ISkinConfig config) {
41          super("top", "navbar");
42          this.setTheme(config.getAttributeValue(COMPONENT, "theme", String.class, "light"));
43          this.setBackground(config.getAttributeValue(COMPONENT, "background", String.class, "light"));
44          this.setCssClass(config.getAttributeValue(COMPONENT, "cssClass", String.class, null));
45          this.withFlatten(config.getAttributeValue(COMPONENT, "flatten", Boolean.class, false))
46                  .withNumberItems(config.getAttributeValue(COMPONENT, "numberItems", Integer.class, -1))
47                  .withEnabled(true);
48          if (isEnabled()) {
49              this.addCssOptions("m-toc-top-enabled");
50          }
51      }
52  
53      /**
54       * Gets the indicating whether is flatten.
55       *
56       * @return Returns {@code true} if is flatten, otherwise {@code false}.
57       */
58      public boolean isFlatten() {
59          return flatten;
60      }
61  
62      /**
63       * Sets the indicating whether is flatten.
64       *
65       * @param flatten
66       *            {@code true} to flat.
67       * @return Returns the fluent instance.
68       */
69      protected TocTopBar withFlatten(final boolean flatten) {
70          this.flatten = flatten;
71          return self();
72      }
73  
74      /**
75       * @return Returns the number of items to display.
76       */
77      public int getNumberItems() {
78          return numberItems;
79      }
80  
81      /**
82       * Sets the number of items to display.
83       *
84       * @param numberItems
85       *            -1 or the number of items to display
86       * @return Returns the fluent instance.
87       */
88      protected TocTopBar withNumberItems(final int numberItems) {
89          this.numberItems = numberItems;
90          return self();
91      }
92  
93  }
