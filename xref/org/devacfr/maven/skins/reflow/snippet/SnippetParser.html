16  package org.devacfr.maven.skins.reflow.snippet;
17  
18  import static java.util.Objects.requireNonNull;
19  
20  import java.io.IOException;
21  import java.util.ArrayList;
22  import java.util.Iterator;
23  
24  import javax.annotation.Nonnull;
25  
26  import org.devacfr.maven.skins.reflow.ISkinConfig;
27  import org.devacfr.maven.skins.reflow.snippet.Processor.ShortcodeProcessor;
28  import org.devacfr.maven.skins.reflow.snippet.Processor.WebComponentProcessor;
29  import org.jsoup.Jsoup;
30  import org.jsoup.nodes.Document;
31  import org.jsoup.nodes.Element;
32  import org.jsoup.select.Elements;
33  
34  /**
35   * @author Christophe Friederich
36   * @version 2.4
37   */
38  public class SnippetParser {
39  
40      /** */
41      private final ComponentResolver resolver;
42  
43      /** */
44      private final ArrayList<ComponentToken> stack;
45  
46      /** */
47      private Iterator<Element> it;
48  
49      /** */
50      private Processor state = null;
51  
52      /** */
53      private final WebComponentProcessor webComponentProcessor;
54  
55      /** */
56      private final ShortcodeProcessor shortcodeProcessor;
57  
58      /** */
59      private final SnippetContext snippetContext;
60  
61      /** */
62      private ComponentToken currentToken;
63  
64      public SnippetParser() {
65          resolver = new ComponentResolver();
66          stack = new ArrayList<>(32);
67          snippetContext = new SnippetContext(this);
68          webComponentProcessor = new WebComponentProcessor(this);
69          shortcodeProcessor = new ShortcodeProcessor(this);
70      }
71  
72      public SnippetParser insertResourcePath(final String path, final int index) {
73          snippetContext.insertResourcePath(path, index);
74          return this;
75      }
76  
77      public SnippetParser addResourcePath(final String path) {
78          snippetContext.addResourcePath(path);
79          return this;
80      }
81  
82      public SnippetContext parse(@Nonnull final ISkinConfig config, @Nonnull final String htmlSource)
83              throws IOException {
84          requireNonNull(config);
85          requireNonNull(htmlSource);
86  
87          snippetContext.reset();
88          snippetContext.setConfig(config);
89  
90          // find all snippets
91          final Document doc = resolver.normalize(Jsoup.parse(htmlSource));
92  
93          final Elements elements = resolver.collect(doc);
94  
95          for (it = elements.iterator(); it.hasNext();) {
96              try {
97                  parse();
98              } catch (final Exception ex) {
99                  throw new SnippetParseException(
100                         "error on parse token " + currentToken + " when generate file " + config.getFileId(), ex);
101             }
102         }
103         snippetContext.setHtmlSource(doc.html());
104         return snippetContext;
105     }
106 
107     protected void parse() {
108         if (!it.hasNext()) {
109             throw new SnippetParseException("EOF");
110         }
111         final Element element = it.next();
112         currentToken = resolver.create(element);
113         if (currentToken == null) {
114             throw new SnippetParseException("unknown component: " + element);
115         }
116         switch (currentToken.type()) {
117             case webComponent:
118                 state = webComponentProcessor;
119                 break;
120             case shortcode:
121                 state = shortcodeProcessor;
122                 break;
123             default:
124                 throw new SnippetParseException("unknown token type " + currentToken.type());
125         }
126         parse(currentToken);
127         currentToken = null;
128     }
129 
130     protected void parse(final ComponentToken token) {
131         state.parse(token);
132     }
133 
134     protected ComponentToken currentToken() {
135         final int size = stack.size();
136         return size > 0 ? stack.get(size - 1) : null;
137     }
138 
139     protected ComponentToken pop() {
140         final int size = stack.size();
141         return stack.remove(size - 1);
142     }
143 
144     protected void push(final ComponentToken element) {
145         stack.add(element);
146     }
147 
148     protected SnippetContext getSnippetContext() {
149         return snippetContext;
150     }
151 
152 }
