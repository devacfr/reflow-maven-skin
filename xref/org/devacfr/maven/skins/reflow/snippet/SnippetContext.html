134     @Nonnull
135     SnippetComponent<?> create(@Nonnull final Element element,
136         @Nonnull final ComponentToken startToken,
137         @Nullable final ComponentToken endToken) {
138         requireNonNull(element);
139         requireNonNull(startToken);
140         final SnippetComponent<?> component = SnippetComponent.createSnippet(element, null, startToken.type());
141         addComponent(component);
142         recurciveCreateComponent(element, component);
143         return component;
144     }
145 
146     @Nonnull
147     private SnippetComponent<?> create(@Nonnull final Element element, final Component<?> commponent) {
148         requireNonNull(element);
149         Type type = null;
150         if (element.hasAttr("shortcode")) {
151             type = Type.shortcode;
152         } else if (element.hasAttr("webcomponent")) {
153             type = Type.webComponent;
154         } else {
155             throw new SnippetParseException("Unknown snippet element");
156         }
157         final SnippetComponent<?> component = SnippetComponent.createSnippet(element, commponent, type);
158         addComponent(component);
159         recurciveCreateComponent(element, component);
160         return component;
161     }
162 
163     private void recurciveCreateComponent(@Nonnull final Node element, final Component<?> parent) {
164         element.childNodes().forEach((child) -> {
165             Component<?> component = null;
166             // accept textnode not empty as component.
167             if (child instanceof TextNode && ((TextNode) child).text().length() > 1) {
168                 component = Component.createComponent(child, parent);
169             } else if (child instanceof Element) {
170                 final Element el = (Element) child;
171                 if (ComponentResolver.isSnippet(el)) {
172                     component = create(el, parent);
173                 } else {
174                     component = Component.createComponent(el, parent);
175                     recurciveCreateComponent(el, component);
176                 }
177             }
178             if (component != null) {
179                 parent.addChild(component);
180             }
181         });
182     }
183 
184     protected void render(final SnippetComponent<?> component) {
185         traverseTee(component, (c) -> {
186             if (c instanceof SnippetComponent) {
187                 ((SnippetComponent<?>) c).render(this);
188             }
189         });
190         component.render(this);
191     }
192 
193     private void traverseTee(final Component<?> component, final Consumer<Component<?>> consumer) {
194         final Consumer<Component<?>> traverse = (c) -> traverseTee(c, consumer);
195         component.getChildren().forEach(consumer.andThen(traverse));
196     }
197 
198     protected String renderComponent(final SnippetComponent<?> component) {
199         final StringWriter writer = new StringWriter();
200         mergeTemplate(component, writer);
201         return writer.toString();
202     }
203 
204     protected void mergeTemplate(final SnippetComponent<?> component, final Writer writer) {
205         boolean found = false;
206 
207         for (final String path : this.snippetPaths) {
208 
209             final String filePath = path + '/' + component.getName() + ".vm";
210             if (Velocity.resourceExists(filePath)) {
211                 found = true;
212                 final Context context = createVelocityContext();
213                 context.put("snippet", component);
214                 context.put("snippetPath", filePath);
215                 context.put("config", this.config);
216                 context.put("velocity", Velocity.class);
217                 context.put("decoration", this.config.getDecoration());
218                 // Use config option <absoluteResourceURL>http://mysite.com/</absoluteResourceURL>
219                 context.put("resourcePath", this.config.getResourcePath());
220 
221                 Velocity.mergeTemplate("META-INF/skin/snippets/_snippet.vm",
222                     RuntimeSingleton.getString(RuntimeConstants.INPUT_ENCODING, RuntimeConstants.ENCODING_DEFAULT),
223                     context,
224                     writer);
225             }
226 
227         }
228         if (!found) {
229             LOGGER.warn("The snippet '{}' template doesn't exist", component.getName());
230         }
231 
232     }

234     /**
235      * Creates a Velocity Context with all generic tools configured wit the site rendering context.
236      *
237      * @return a Velocity tools managed context
238      */
239     protected Context createVelocityContext() {
240         return toolManager.createContext();
241     }
242 
243     /**
244      * @return
245      */
246     protected ToolManager createToolManaged() {
247 
248         final EasyFactoryConfiguration config = new EasyFactoryConfiguration(false);
249         config.property("safeMode", Boolean.FALSE);
250         config.toolbox(Scope.REQUEST)
251                 .tool(ContextTool.class)
252                 .tool(LinkTool.class)
253                 .tool(LoopTool.class)
254                 .tool(RenderTool.class);
255         config.toolbox(Scope.APPLICATION)
256                 .tool(AlternatorTool.class)
257                 .tool(ClassTool.class)
258                 .tool(ComparisonDateTool.class)
259                 .tool(ConversionTool.class)
260                 .tool(DisplayTool.class)
261                 .tool(EscapeTool.class)
262                 .tool(FieldTool.class)
263                 .tool(MathTool.class)
264                 .tool(NumberTool.class)
265                 .tool(ResourceTool.class)
266                 .tool(SortTool.class)
267                 .tool(XmlTool.class)
268                 .tool(URITool.class)
269                 .tool(HtmlTool.class);
270 
271         final ToolManager manager = new ToolManager(false, false);
272         manager.configure(config);
273         return manager;
274     }
