43  /**
44   * An Apache Velocity tool that simplifies retrieval of custom configuration values for a Maven Site.
45   * <p>
46   * The tool is configured to access Maven site configuration of a skin inside {@code <custom>} element of site
47   * descriptor. It supports global properties (defined at skin level) and per-page properties (defined in
48   * {@code <page><mypage>} element). The per-page properties override the global ones.
49   * </p>
50   * <p>
51   * A sample configuration would be like that:
52   * </p>
53   *
54   * <pre>
55   * {@code
56   * <custom>
57   *   <reflowSkin>
58   *     <prop1>value1</prop1>
59   *     <prop2>
60   *       <prop21>value2</prop21>
61   *     </prop2>
62   *     <pages>
63   *       <mypage project="myproject">
64   *         <prop1>override value1</prop1>
65   *       </mypage>
66   *     </pages>
67   *   </reflowSkin>
68   * </custom>
69   * }
70   * </pre>
71   * <p>
72   * To get the value of {@code prop1}, one would simply use {@code $config.prop1}. This would return "override value1".
73   * Then {@code $config.prop2} would return "value2" - the global value.
74   * </p>
75   * <p>
76   * The tool allows querying the value easily, falling back from page to global configuration to {@code null}, if none is
77   * available. It also provides convenience accessors for common values.
78   * </p>
79   * <p>
80   * Note
81   * </p>
82   *
83   * @author Andrius Velykis
84   * @author Christophe Friederich
85   * @since 1.0
86   */

132     /**
133      * {@inheritDoc}
134      *
135      * @see SafeConfig#configure(ValueParser)
136      */
137     @Override
138     protected void configure(final ValueParser values) {
139         final String altkey = values.getString("key");
140         if (altkey != null) {
141             setKey(altkey);
142         }
143 
144         // allow changing skin key in the configuration
145         final String altSkinKey = values.getString("skinKey");
146         if (altSkinKey != null) {
147             this.skinKey = altSkinKey;
148         }
149 
150         // retrieve the decoration model from Velocity context
151         final Object vc = values.get("velocityContext");
152 
153         if (!(vc instanceof ToolContext)) {
154             return;
155         }
156 
157         this.velocityContext = (ToolContext) vc;
158 
159         final Object projectObj = velocityContext.get("project");
160         if (projectObj instanceof MavenProject) {
161             this.project = (MavenProject) projectObj;
162             final String artifactId = project.getArtifactId();
163             // use artifactId "sluggified" as the projectId
164             projectId = HtmlTool.slug(artifactId);
165         }
166 
167         // calculate the page ID from the current file name
168         final Object currentFileObj = velocityContext.get("currentFileName");
169         if (currentFileObj instanceof String) {
170             fileId = slugFilename((String) currentFileObj);
171         }
172 
173         final Object decorationObj = velocityContext.get("decoration");
174 
175         if (!(decorationObj instanceof DecorationModel)) {
176             return;
177         }
178 
179         this.decoration = (DecorationModel) decorationObj;
180         final Object customObj = decoration.getCustom();
181 
182         if (!(customObj instanceof Xpp3Dom)) {
183             return;
184         }
185 
186         // Now that we have the custom node, get the global properties
187         // under the skin tag
188         final Xpp3Dom customNode = (Xpp3Dom) customObj;
189         Xpp3Dom skinNode = customNode.getChild(skinKey);
190         final String namespaceKey = ":" + skinKey;
191 
192         if (skinNode == null) {
193             // try searching with any namespace
194             for (final Xpp3Dom child : customNode.getChildren()) {
195                 if (child.getName().endsWith(namespaceKey)) {
196                     skinNode = child;
197                     break;
198                 }
199             }
200         }
201 
202         if (skinNode != null) {
203             globalProperties = skinNode;
204 
205             if (skinNode.getName().endsWith(namespaceKey)) {
206                 // extract the namespace (including the colon)
207                 namespace = skinNode.getName().substring(0, skinNode.getName().length() - namespaceKey.length() + 1);
208             }
209 
210             // for page properties, retrieve the file name and drop the `.html`
211             // extension - this will be used, i.e. `index` instead of `index.html`
212             final Xpp3Dom pagesNode = Xpp3Utils.getFirstChild(skinNode, "pages", namespace);
213             if (pagesNode != null) {
214 
215                 // Get the page for the file
216                 Xpp3Dom page = Xpp3Utils.getFirstChild(pagesNode, fileId, namespace);
217 
218                 // Now check if the project artifact ID is set, and if so, if it matches the
219                 // current project. This allows preventing accidental reuse of parent page
220                 // configs in children modules
221                 if (page != null && projectId != null) {
222                     final String pageProject = page.getAttribute("project");
223                     if (pageProject != null && !projectId.equals(pageProject)) {
224                         // project ID indicated, and is different - do not use the config
225                         page = null;
226                     }
227                 }
228 
229                 if (page != null) {
230                     pageProperties = page;
231                 }
232 
233             }
234 
235             // Config option <localResources>true</localResources> to force CDN-less Bootstrap & jQuery
236             this.velocityContext.put("localResources", is("localResources"));
237             // Use config option <absoluteResourceURL>http://mysite.com/</absoluteResourceURL>
238             this.velocityContext.put("resourcePath", getResourcePath());
239 
240             this.context = Context.buildContext(this);
241         }
242         if (LOGGER.isDebugEnabled()) {
243             LOGGER.debug("Current Filename: {}", currentFileObj);
244             LOGGER.debug("Project id: {}", projectId);
245             LOGGER.debug("File id: {}", fileId);
246             LOGGER.debug("Context: {}", this.context);
247             LOGGER.debug("---------------------------------------------------");
248         }
249     }

270     /**
271      * Default accessor for config properties. Instead of using {@code $config.get("myproperty")}, one can utilise
272      * Velocity fallback onto the default getter and use {@code $config.myproperty}.
273      *
274      * @param property
275      *            the property of interest
276      * @return configuration node if found in the following sequence:
277      *         <ol>
278      *         <li>In page configuration</li>
279      *         <li>In global configuration</li>
280      *         <li>{@code null} otherwise</li>
281      *         </ol>
282      * @since 1.0
283      */
284     public Xpp3Dom get(final String property) {
285 
286         // first try page properties
287         Xpp3Dom propNode = Xpp3Utils.getFirstChild(pageProperties, property, namespace);
288         if (propNode == null) {
289             // try global
290             propNode = Xpp3Utils.getFirstChild(globalProperties, property, namespace);
291         }
292 
293         return propNode;
294     }

297      * Retrieves the text value of the given {@code property}, e.g. as in {@code <myprop>value</myprop>}.
298      *
299      * @param property
300      *            the property of interest
301      * @return the configuration value if found in page or globally, {@code null} otherwise.
302      * @see #get(String)
303      * @since 1.0
304      */
305     public String value(final String property) {
306 
307         final Xpp3Dom propNode = get(property);
308 
309         if (propNode == null) {
310             // not found
311             return null;
312         }
313 
314         return propNode.getValue();
315     }

317     /**
318      * Gets the text value of the given {@code property}.
319      *
320      * @param property
321      *            the property to use
322      * @param targetType
323      *            the returned target type use to convert value.
324      * @param defaultValue
325      *            the default value used if property doesn't exist.
326      * @return Returns a converted value of the given {@code property}.
327      * @since 2.0
328      * @param <T>
329      *            the type of returned object.
330      */
331     @SuppressWarnings("unchecked")
332     public <T> T getPropertyValue(final String property, final Class<T> targetType, final T defaultValue) {
333         final String value = value(property);
334         if (value == null) {
335             return defaultValue;
336         }
337         Object returnedValue = value;
338         if (targetType.isAssignableFrom(Boolean.class)) {
339             returnedValue = Boolean.valueOf(value);
340         } else if (targetType.isAssignableFrom(Integer.class)) {
341             returnedValue = Integer.valueOf(value);
342         } else if (targetType.isAssignableFrom(Long.class)) {
343             returnedValue = Long.valueOf(value);
344         }
345         return (T) returnedValue;
346     }

361     /**
362      * Gets the attribute value of the given {@code attribute} of {@code property}.
363      *
364      * @param property
365      *            the property to use
366      * @param attribute
367      *            the attribute to use.
368      * @param targetType
369      *            the returned target type use to convert value.
370      * @param defaultValue
371      *            the default value used if property doesn't exist.
372      * @return Returns a converted value of the given {@code property}.
373      * @since 2.0
374      * @param <T>
375      *            the type of returned object.
376      */
377     @SuppressWarnings("unchecked")
378     public <T> T getAttributeValue(final String property,
379         final String attribute,
380         final Class<T> targetType,
381         final T defaultValue) {
382         Xpp3Dom element = get(property);
383         if (element == null) {
384             return defaultValue;
385         }
386         String value = element.getAttribute(attribute);
387         if (value == null) {
388             return defaultValue;
389         }
390 
391         if ("inherit".equals(value)) {
392             element = Xpp3Utils.getFirstChild(globalProperties, property, namespace);
393             if (LOGGER.isDebugEnabled()) {
394                 LOGGER.debug("Inherit value property '{}': {}", property, element);
395             }
396         }
397         if (element == null) {
398             return defaultValue;
399         }
400         value = element.getAttribute(attribute);
401         if (value == null) {
402             return defaultValue;
403         }
404 
405         Object returnedValue = value;
406         if (targetType.isAssignableFrom(Boolean.class)) {
407             returnedValue = Boolean.valueOf(value);
408         } else if (targetType.isAssignableFrom(Integer.class)) {
409             returnedValue = Integer.valueOf(value);
410         } else if (targetType.isAssignableFrom(Long.class)) {
411             returnedValue = Long.valueOf(value);
412         }
413         return (T) returnedValue;
414     }

579     /**
580      * @param href
581      *            link to relative.
582      * @return Returns Relativizes the link.
583      */
584     @Nullable
585     public String relativeLink(final String href) {
586         if (href == null) {
587             return null;
588         }
589         if (isExternalLink(href)) {
590             return href;
591         }
592         final String relativePath = (String) velocityContext.get("relativePath");
593         String relativeLink = PathTool.calculateLink(href, relativePath);
594         relativeLink = relativeLink.replaceAll("\\\\", "/");
595         if (Strings.isNullOrEmpty(relativeLink)) {
596             relativeLink = "./";
597         }
598         // Attempt to normalise the relative link - this is useful for active link
599         // calculations and better relative links for subdirectories.
600         //
601         // The issue is particularly visible with pages in subdirectories,
602         // so that if you are in <root>/dev/index.html, the relative menu link to
603         // the _same_ page would likely be ../dev/index.html instead of '' or
604         // 'index.html'.
605         final String currentFileLoc = getCurrentFileLocation();
606         final String absoluteLink = URITool.toURI(currentFileLoc).resolve(relativeLink).normalize().toString();
607         if (currentFileLoc.equals(absoluteLink)) {
608             // for matching link, use empty relative link
609             relativeLink = StringUtils.EMPTY;
610         } else {
611             // relativize the absolute link based on current directory
612             // (uses Maven project link relativization)
613             final String currentFileDir = PathTool.getDirectoryComponent(currentFileLoc);
614             relativeLink = URITool.relativizeLink(currentFileDir, absoluteLink);
615         }
616         if (LOGGER.isDebugEnabled()) {
617             LOGGER.debug("-- Relative Link ----------------------------------");
618             LOGGER.debug("link: {}", href);
619             LOGGER.debug("currentFileLoc: {}", currentFileLoc);
620             LOGGER.debug("absoluteLink: {}", absoluteLink);
621             LOGGER.debug("relativeLink: {}", relativeLink);
622             LOGGER.debug("---------------------------------------------------");
623         }
624         return relativeLink;
625     }

627     /**
628      * @param url
629      *            a url.
630      * @return Returns {@code true} whether the link is a external link to the site.
631      */
632     public boolean isExternalLink(final String url) {
633         if (url == null) {
634             return false;
635         }
636         final String absoluteResourceURL = this.value("absoluteResourceURL");
637         if (!Strings.isNullOrEmpty(absoluteResourceURL) && url.startsWith(absoluteResourceURL)) {
638             return false;
639         }
640         return url.toLowerCase().startsWith("http:/") || url.toLowerCase().startsWith("https:/")
641                 || url.toLowerCase().startsWith("ftp:/") || url.toLowerCase().startsWith("mailto:")
642                 || url.toLowerCase().startsWith("file:/") || url.toLowerCase().indexOf("://") != -1;
643     }

645     /**
646      *
647      */
648     public boolean isActiveLink(final String href) {
649         final String alignedFileName = (String) velocityContext.get("alignedFileName");
650         if (href == null) {
651             return false;
652         }
653         // either empty link (pointing to a page), or if the current file is index.html,
654         // the link may point to the whole directory
655         return Strings.isNullOrEmpty(href) || alignedFileName.endsWith("index.html") && ".".equals(href);
656     }

670     /**
671      * Converts a filename to pageId format.
672      *
673      * @param fileName
674      *            the filename to convert
675      * @return Returns a {@link String} representing the pageId of {@code filename}.
676      */
677     @Nullable
678     public static String slugFilename(@Nullable final String fileName) {
679         if (fileName == null) {
680             return null;
681         }
682         String currentFile = fileName;
683 
684         // drop the extension
685         final int lastDot = currentFile.lastIndexOf(".");
686         if (lastDot >= 0) {
687             currentFile = currentFile.substring(0, lastDot);
688         }
689 
690         // get the short ID (in case of nested files)
691         // String fileName = new File(currentFile).getName();
692         // fileShortId = HtmlTool.slug(fileName);
693 
694         // full file ID includes the nested dirs
695         // replace nesting "/" with "-"
696         return HtmlTool.slug(currentFile.replace("/", "-").replace("\\", "-"));
697     }

699     /**
700      * @return Returns a {@link String} representing the relative path to root site.
701      */
702     @Nonnull
703     private String getResourcePath() {
704         final String absoluteResourceURL = this.value("absoluteResourceURL");
705         String projectUrl = getProjectLocation();
706         final String currentFileName = (String) velocityContext.get("currentFileName");
707         if (!Strings.isNullOrEmpty(projectUrl) && currentFileName != null) {
708             if (projectUrl.charAt(projectUrl.length() - 1) != '/') {
709                 projectUrl += '/';
710             }
711             final String currentFileDir = URITool.toURI(projectUrl).resolve(currentFileName).resolve(".").toString();
712             return URITool.relativizeLink(currentFileDir, absoluteResourceURL);
713         }
714         return (String) velocityContext.get("relativePath");
715     }

717     /**
718      * @return Returns new instance of {@link URLRebaser}.
719      */
720     @Nonnull
721     private URLRebaser createURLRebaser() {
722         String childBaseUrl = this.getProject().getUrl();
723         if (Strings.isNullOrEmpty(childBaseUrl)) {
724             childBaseUrl = null;
725         }
726         final String relativePath = getResourcePath();
727         String parentBaseUrl = relativePath;
728         if (childBaseUrl != null && childBaseUrl.length() > 0) {
729             if (childBaseUrl.charAt(childBaseUrl.length() - 1) != '/') {
730                 childBaseUrl += '/';
731             }
732             final URI child = URI.create(childBaseUrl);
733             parentBaseUrl = child.resolve(relativePath).normalize().toString();
734         }
735         if (LOGGER.isDebugEnabled()) {
736             LOGGER.debug("parentBaseUrl: {}", parentBaseUrl);
737             LOGGER.debug("childBaseUrl: {}", childBaseUrl);
738             LOGGER.debug("relativePath: {}", relativePath);
739         }
740 
741         return new URLRebaser(parentBaseUrl, childBaseUrl);
742     }
