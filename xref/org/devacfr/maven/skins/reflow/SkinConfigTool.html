41  /**
42   * An Apache Velocity tool that simplifies retrieval of custom configuration values for a Maven Site.
43   * <p>
44   * The tool is configured to access Maven site configuration of a skin inside {@code <custom>} element of site
45   * descriptor. It supports global properties (defined at skin level) and per-page properties (defined in
46   * {@code <page><mypage>} element). The per-page properties override the global ones.
47   * </p>
48   * <p>
49   * A sample configuration would be like that:
50   * </p>
51   *
52   * <pre>
53   * {@code
54   * <custom>
55   *   <reflowSkin>
56   *     <prop1>value1</prop1>
57   *     <prop2>
58   *       <prop21>value2</prop21>
59   *     </prop2>
60   *     <pages>
61   *       <mypage project="myproject">
62   *         <prop1>override value1</prop1>
63   *       </mypage>
64   *     </pages>
65   *   </reflowSkin>
66   * </custom>
67   * }
68   * </pre>
69   * <p>
70   * To get the value of {@code prop1}, one would simply use {@code $config.prop1}. This would return "override value1".
71   * Then {@code $config.prop2} would return "value2" - the global value.
72   * </p>
73   * <p>
74   * The tool allows querying the value easily, falling back from page to global configuration to {@code null}, if none is
75   * available. It also provides convenience accessors for common values.
76   * </p>
77   * <p>
78   * Note
79   * </p>
80   *
81   * @author Andrius Velykis
82   * @author Christophe Friederich
83   * @since 1.0
84   */

130     /**
131      * {@inheritDoc}
132      *
133      * @see SafeConfig#configure(ValueParser)
134      */
135     @Override
136     protected void configure(final ValueParser values) {
137         final String altkey = values.getString("key");
138         if (altkey != null) {
139             setKey(altkey);
140         }
141 
142         // allow changing skin key in the configuration
143         final String altSkinKey = values.getString("skinKey");
144         if (altSkinKey != null) {
145             this.skinKey = altSkinKey;
146         }
147 
148         // retrieve the decoration model from Velocity context
149         final Object vc = values.get("velocityContext");
150 
151         if (!(vc instanceof ToolContext)) {
152             return;
153         }
154 
155         this.velocityContext = (ToolContext) vc;
156 
157         final Object projectObj = velocityContext.get("project");
158         if (projectObj instanceof MavenProject) {
159             this.project = (MavenProject) projectObj;
160             final String artifactId = project.getArtifactId();
161             // use artifactId "sluggified" as the projectId
162             projectId = HtmlTool.slug(artifactId);
163         }
164 
165         // calculate the page ID from the current file name
166         final String currentFileObj = getCurrentFileName();
167         fileId = slugFilename(currentFileObj);
168 
169         final Object decorationObj = velocityContext.get("decoration");
170 
171         if (!(decorationObj instanceof DecorationModel)) {
172             return;
173         }
174 
175         this.decoration = (DecorationModel) decorationObj;
176         final Object customObj = decoration.getCustom();
177 
178         if (!(customObj instanceof Xpp3Dom)) {
179             return;
180         }
181 
182         // Now that we have the custom node, get the global properties
183         // under the skin tag
184         final Xpp3Dom customNode = (Xpp3Dom) customObj;
185         Xpp3Dom skinNode = customNode.getChild(skinKey);
186         final String namespaceKey = ":" + skinKey;
187 
188         if (skinNode == null) {
189             // try searching with any namespace
190             for (final Xpp3Dom child : customNode.getChildren()) {
191                 if (child.getName().endsWith(namespaceKey)) {
192                     skinNode = child;
193                     break;
194                 }
195             }
196         }
197 
198         if (skinNode != null) {
199             globalProperties = skinNode;
200 
201             if (skinNode.getName().endsWith(namespaceKey)) {
202                 // extract the namespace (including the colon)
203                 namespace = skinNode.getName().substring(0, skinNode.getName().length() - namespaceKey.length() + 1);
204             }
205 
206             // for page properties, retrieve the file name and drop the `.html`
207             // extension - this will be used, i.e. `index` instead of `index.html`
208             final Xpp3Dom pagesNode = Xpp3Utils.getFirstChild(skinNode, "pages", namespace);
209             if (pagesNode != null) {
210 
211                 // Get the page for the file
212                 Xpp3Dom page = Xpp3Utils.getFirstChild(pagesNode, fileId, namespace);
213 
214                 // Now check if the project artifact ID is set, and if so, if it matches the
215                 // current project. This allows preventing accidental reuse of parent page
216                 // configs in children modules
217                 if (page != null && projectId != null) {
218                     final String pageProject = page.getAttribute("project");
219                     if (pageProject != null && !projectId.equals(pageProject)) {
220                         // project ID indicated, and is different - do not use the config
221                         page = null;
222                     }
223                 }
224 
225                 if (page != null) {
226                     pageProperties = page;
227                 }
228 
229             }
230 
231             // Config option <localResources>true</localResources> to force CDN-less Bootstrap & jQuery
232             this.velocityContext.put("localResources", is("localResources"));
233             // Use config option <absoluteResourceURL>http://mysite.com/</absoluteResourceURL>
234             this.velocityContext.put("resourcePath", getResourcePath());
235 
236             this.context = Context.buildContext(this);
237         }
238         if (LOGGER.isDebugEnabled()) {
239             LOGGER.debug("Current Filename: {}", currentFileObj);
240             LOGGER.debug("Project id: {}", projectId);
241             LOGGER.debug("File id: {}", fileId);
242             LOGGER.debug("Context: {}", this.context);
243             LOGGER.debug("Namespace: {}", this.namespace);
244             LOGGER.debug("---------------------------------------------------");
245         }
246     }

345      * Gets the text value of the given {@code property}.
346      *
347      * @param property
348      *            the property to use
349      * @param targetType
350      *            the returned target type use to convert value.
351      * @param defaultValue
352      *            the default value used if property doesn't exist.
353      * @return Returns a converted value of the given {@code property}.
354      * @since 2.0
355      * @param <T>

358     @Override
359     @SuppressWarnings("unchecked")
360     @Nullable
361     public <T> T getPropertyValue(@Nonnull final String property,
362         @Nonnull final Class<T> targetType,
363         @Nullable final T defaultValue) {
364         requireNonNull(property, "property is required");
365         requireNonNull(targetType, "targetType is required");
366         final String value = value(property);
367         if (value == null) {
368             return defaultValue;
369         }
370         Object returnedValue = value;
371         if (targetType.isAssignableFrom(Boolean.class)) {
372             returnedValue = Boolean.valueOf(value);
373         } else if (targetType.isAssignableFrom(Integer.class)) {
374             returnedValue = Integer.valueOf(value);
375         } else if (targetType.isAssignableFrom(Long.class)) {
376             returnedValue = Long.valueOf(value);
377         }
378         return (T) returnedValue;
379     }

394     /**
395      * Gets the attribute value of the given {@code attribute} of {@code property}.
396      *
397      * @param property
398      *            the property to use
399      * @param attribute
400      *            the attribute to use.
401      * @param targetType
402      *            the returned target type use to convert value.
403      * @param defaultValue
404      *            the default value used if property doesn't exist.
405      * @return Returns a converted value of the given {@code property}.
406      * @since 2.0
407      * @param <T>
408      *            the type of returned object.
409      */
410     @Override
411     @SuppressWarnings("unchecked")
412     @Nullable
413     public <T> T getAttributeValue(@Nonnull final String property,
414         @Nonnull final String attribute,
415         @Nonnull final Class<T> targetType,
416         @Nullable final T defaultValue) {
417         requireNonNull(property, "property is required");
418         requireNonNull(attribute, "attribute is required");
419         requireNonNull(targetType, "targetType is required");
420 
421         Xpp3Dom element = get(property);
422         if (element == null) {
423             return defaultValue;
424         }
425         String value = element.getAttribute(attribute);
426         if (value == null) {
427             return defaultValue;
428         }
429 
430         if ("inherit".equals(value) || Strings.isNullOrEmpty(value)) {
431             element = Xpp3Utils.getFirstChild(globalProperties, property, namespace);
432             if (LOGGER.isDebugEnabled()) {
433                 LOGGER.debug("Inherit value property '{}': {}", property, element);
434             }
435         }
436         if (element == null) {
437             return defaultValue;
438         }
439         value = element.getAttribute(attribute);
440         if (value == null) {
441             return defaultValue;
442         }
443 
444         Object returnedValue = value;
445         if (targetType.isAssignableFrom(Boolean.class)) {
446             returnedValue = Boolean.valueOf(value);
447         } else if (targetType.isAssignableFrom(Integer.class)) {
448             returnedValue = Integer.valueOf(value);
449         } else if (targetType.isAssignableFrom(Long.class)) {
450             returnedValue = Long.valueOf(value);
451         }
452         return (T) returnedValue;
453     }

654     /**
655      * {@inheritDoc}
656      */
657     @Override
658     public String relativeLink(final String href) {
659         if (href == null) {
660             return null;
661         }
662         if (isExternalLink(href)) {
663             return href;
664         }
665         final String relativePath = (String) velocityContext.get("relativePath");
666         String relativeLink = PathTool.calculateLink(href, relativePath);
667         relativeLink = relativeLink.replaceAll("\\\\", "/");
668         if (Strings.isNullOrEmpty(relativeLink)) {
669             relativeLink = "./";
670         }
671         // Attempt to normalise the relative link - this is useful for active link
672         // calculations and better relative links for subdirectories.
673         //
674         // The issue is particularly visible with pages in subdirectories,
675         // so that if you are in <root>/dev/index.html, the relative menu link to
676         // the _same_ page would likely be ../dev/index.html instead of '' or
677         // 'index.html'.
678         final String currentFileLoc = getCurrentFileLocation();
679         final String absoluteLink = URITool.toURI(currentFileLoc).resolve(relativeLink).normalize().toString();
680         if (currentFileLoc.equals(absoluteLink)) {
681             // for matching link, use empty relative link
682             relativeLink = StringUtils.EMPTY;
683         } else {
684             // relativize the absolute link based on current directory
685             // (uses Maven project link relativization)
686             final String currentFileDir = PathTool.getDirectoryComponent(currentFileLoc);
687             relativeLink = URITool.relativizeLink(currentFileDir, absoluteLink);
688         }
689         if (LOGGER.isDebugEnabled()) {
690             LOGGER.debug("-- Relative Link ----------------------------------");
691             LOGGER.debug("link: {}", href);
692             LOGGER.debug("currentFileLoc: {}", currentFileLoc);
693             LOGGER.debug("absoluteLink: {}", absoluteLink);
694             LOGGER.debug("relativeLink: {}", relativeLink);
695             LOGGER.debug("---------------------------------------------------");
696         }
697         return relativeLink;
698     }

700     /**
701      * @param url
702      *            a url.
703      * @return Returns {@code true} whether the link is a external link to the site.
704      */
705     public boolean isExternalLink(final String url) {
706         if (url == null) {
707             return false;
708         }
709         final String absoluteResourceURL = this.value("absoluteResourceURL");
710         if (!Strings.isNullOrEmpty(absoluteResourceURL) && url.startsWith(absoluteResourceURL)) {
711             return false;
712         }
713         return url.toLowerCase().startsWith("http:/") || url.toLowerCase().startsWith("https:/")
714                 || url.toLowerCase().startsWith("ftp:/") || url.toLowerCase().startsWith("mailto:")
715                 || url.toLowerCase().startsWith("file:/") || url.toLowerCase().indexOf("://") != -1;
716     }

718     /**
719      * {@inheritDoc}
720      */
721     @Override
722     public boolean isActiveLink(@Nullable final String href) {
723         final String alignedFileName = (String) velocityContext.get("alignedFileName");
724         if (href == null) {
725             return false;
726         }
727         // either empty link (pointing to a page), or if the current file is index.html,
728         // the link may point to the whole directory
729         return Strings.isNullOrEmpty(href) || alignedFileName.endsWith("index.html") && ".".equals(href);
730     }

732     /**
733      * Converts a filename to pageId format.
734      *
735      * @param fileName
736      *            the filename to convert
737      * @return Returns a {@link String} representing the pageId of {@code filename}.
738      */
739     @Nullable
740     public static String slugFilename(@Nullable final String fileName) {
741         if (fileName == null) {
742             return null;
743         }
744         String currentFile = fileName;
745 
746         // drop the extension
747         final int lastDot = currentFile.lastIndexOf(".");
748         if (lastDot >= 0) {
749             currentFile = currentFile.substring(0, lastDot);
750         }
751 
752         // get the short ID (in case of nested files)
753         // String fileName = new File(currentFile).getName();
754         // fileShortId = HtmlTool.slug(fileName);
755 
756         // full file ID includes the nested dirs
757         // replace nesting "/" with "-"
758         return HtmlTool.slug(currentFile.replace("/", "-").replace("\\", "-"));
759     }

764     @Nonnull
765     private String getResourcePath() {
766         final String absoluteResourceURL = this.value("absoluteResourceURL");
767         String projectUrl = getProjectLocation();
768         final String currentFileName = getCurrentFileName();
769         if (!Strings.isNullOrEmpty(projectUrl) && currentFileName != null) {
770             if (projectUrl.charAt(projectUrl.length() - 1) != '/') {
771                 projectUrl += '/';
772             }
773             final String currentFileDir = URITool.toURI(projectUrl).resolve(currentFileName).resolve(".").toString();
774             return URITool.relativizeLink(currentFileDir, absoluteResourceURL);
775         }
776         return (String) velocityContext.get("relativePath");
777     }
