16  package org.devacfr.maven.skins.reflow.context;
17  
18  import static org.hamcrest.MatcherAssert.assertThat;
19  import static org.hamcrest.Matchers.isA;
20  
21  import java.io.StringReader;
22  
23  import org.apache.maven.doxia.site.decoration.DecorationModel;
24  import org.apache.maven.project.MavenProject;
25  import org.codehaus.plexus.util.xml.Xpp3Dom;
26  import org.codehaus.plexus.util.xml.Xpp3DomBuilder;
27  import org.devacfr.maven.skins.reflow.ISkinConfig;
28  import org.devacfr.maven.skins.reflow.model.Footer;
29  import org.devacfr.maven.skins.reflow.model.Navbar;
30  import org.devacfr.maven.skins.reflow.model.ScrollTop;
31  import org.devacfr.maven.skins.reflow.model.Toc;
32  import org.devacfr.maven.skins.reflow.model.TocSidebar;
33  import org.devacfr.testing.jupiter.MockitoTestCase;
34  import org.junit.jupiter.api.BeforeEach;
35  import org.junit.jupiter.api.Test;
36  import org.mockito.Mock;
37  
38  public class FrameContextTest extends MockitoTestCase {
39  
40      @Mock
41      private ISkinConfig config;
42  
43      @SuppressWarnings("unchecked")
44      @BeforeEach
45      public void setup() {
46          final MavenProject project = new MavenProject();
47          project.setName("reflow");
48          project.setArtifactId("reflow-artifact");
49          when(config.getProject()).thenReturn(project);
50  
51          final DecorationModel decoration = new DecorationModel();
52          when(config.getDecoration()).thenReturn(decoration);
53          when(config.getAttributeValue(any(String.class), any(String.class), any(Class.class), any(Object.class)))
54                  .then(invocation -> invocation.getArguments()[3]);
55          when(config.getPropertyValue(any(String.class), any(Class.class), any(Object.class)))
56                  .then(invocation -> invocation.getArguments()[2]);
57      }
58  
59      @Test
60      public void shouldBuildFrameContext() throws Exception {
61          final String xml = "<reflowSkin><pages>" + //
62                  "<development-documentation type=\"doc\">" + //
63                  "          <menu name=\"Development Documentation\">" + //
64                  "            <item name=\"Contribute\" href=\"dev/contribute.html\" />" + //
65                  "            <item name=\"Code Conventions\" href=\"dev/code-conventions.html\"/>" + //
66                  "            <item name=\"Release Management\" href=\"dev/release-management.html\"/>" + //
67                  "          </menu>" + //
68                  "        </development-documentation>" + //
69                  "</pages></reflowSkin>";
70          final Xpp3Dom globalProperties = Xpp3DomBuilder.build(new StringReader(xml));
71          when(config.getPageProperties()).thenReturn(new Xpp3Dom("dev-contribute"));
72          when(config.getGlobalProperties()).thenReturn(globalProperties);
73          when(config.getFileId()).thenReturn("dev-contribute");
74          when(config.getNamespace()).thenReturn("");
75  
76          final Context<?> context = Context.buildContext(config);
77          assertThat((FrameContext) context, isA(FrameContext.class));
78          final FrameContext frameContext = (FrameContext) context;
79  
80          assertEquals("development-documentation", frameContext.getDocumentParent());
81  
82          final Toc<?> toc = frameContext.getToc();
83          assertNotNull(toc, "toc should be exist");
84          assertThat((TocSidebar) frameContext.getToc(), isA(TocSidebar.class));
85  
86          final TocSidebar tocSidebar = (TocSidebar) toc;
87          assertEquals(Integer.MAX_VALUE, tocSidebar.getLevel());
88          assertEquals("sidebar-light bg-light", tocSidebar.getCssClass());
89          assertEquals("m-toc-sidebar-enabled m-toc-sidebar-expanded m-toc-sidebar-autoexpandable toc-sidebar-fixed",
90              tocSidebar.getCssOptions());
91  
92          final Footer footer = frameContext.getFooter();
93          assertNotNull(footer, "footer should be exist");
94          assertEquals("footer-light bg-light", footer.getCssClass());
95          assertEquals("", footer.getCssOptions());
96  
97          final Navbar navbar = frameContext.getNavbar();
98          assertNotNull(navbar, "Navbar should be exist");
99          assertEquals("navbar-light bg-light", navbar.getCssClass());
100         assertEquals("", navbar.getCssOptions());
101 
102         final ScrollTop scrollTop = frameContext.getScrollTop();
103         assertNotNull(scrollTop, "ScrollTop should be exist");
104         assertEquals("", scrollTop.getCssClass());
105         assertEquals(true, scrollTop.isSmooth());
106         assertEquals("scrolltop-smooth-enabled", scrollTop.getCssOptions());
107 
108     }
109
